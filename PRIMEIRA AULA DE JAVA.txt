import java.util.Scanner;

public class LerNome {
    public static void main(String[] args) {
        // inicializa o scanner
        Scanner leitor = new Scanner(System.in);
        
        // solicita o nome ao usuário
        System.out.print("por favor, digite o seu nome completo: ");
        
        // o método .nextLine() lê a linha inteira (incluindo espaços)
        String nome = leitor.nextLine();
        
        // exibe a mensagem personalizada
        System.out.println("olá, " + nome + "! seja bem-vindo ao mundo java.");
        
        // fechar o scanner
        leitor.close();
    }
}
